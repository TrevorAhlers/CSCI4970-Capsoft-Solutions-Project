import { Component, Input, Output, EventEmitter, OnChanges, ElementRef, ViewChild, AfterViewInit } from '@angular/core';
/**
 * DetailsComponent is responsible for displaying and editing the details of a selected course.
 * It receives data for the course, validates user input, and allows the user to save changes.
 */
@Component({
	selector: 'app-details',
	templateUrl: './details.component.html',
	styleUrls: ['./details.component.scss']
})
export class DetailsComponent implements OnChanges, AfterViewInit {
	@Input() selectedCourse: any;
	@Output() saveClicked = new EventEmitter<any>();
	courseData: { label: string; value: string }[] = [];
	tab = 0;

	@ViewChild('htmlWrapper') htmlWrapperRef!: ElementRef;

	userInput = {
		room: '', meeting1: '', meeting2: '', meeting3: '', meeting4: '',
		crossListings: '', maxEnrollment: '', enrollment: '',
		comments: '', notes1: '', notes2: ''
	};

	errors = {
		room: false,
		meeting1: false,
		maxEnrollment: false,
		enrollment: false
	};
/**
	 * Lifecycle hook that is triggered when the @Input property `selectedCourse` changes.
	 * This method parses the course content and populates the `userInput` object with the course's editable details.
	 */
ngOnChanges(): void {
	if (this.selectedCourse?.content) {
		const parser = new DOMParser();
		const doc = parser.parseFromString(this.selectedCourse.content, 'text/html');
		const pElements = Array.from(doc.querySelectorAll('p'));

		this.courseData = pElements.map(p => {
			const text = p.textContent ?? '';
			const [label, ...rest] = text.split(':');
			return { label: label.trim(), value: rest.join(':').trim() };
		});
	}

	if (this.selectedCourse?.edit) {
		this.userInput = {
			room: this.selectedCourse.edit.room || '',
			meeting1: this.selectedCourse.edit.meeting1 || '',
			crossListings: this.selectedCourse.edit.crossListings || '',
			maxEnrollment: this.selectedCourse.edit.maxEnrollment || '',
			enrollment: this.selectedCourse.edit.enrollment || '',
			comments: this.selectedCourse.edit.comments || '',
			notes1: this.selectedCourse.edit.notes1 || '',
			notes2: this.selectedCourse.edit.notes2 || '',
			meeting2: '', meeting3: '', meeting4: ''
		};
	}

	this.validate();
}
/**
	 * Validates the user input for required fields and populates the `errors` object.
	 * This method is used to ensure that all required fields are filled out before saving.
	 */
validate(): void {
	const r = this.userInput.room.trim();
	this.errors.room = r === '' || (r.length < 3 && r.toLowerCase() !== 'to be announced');
	this.errors.meeting1 = this.userInput.meeting1.trim() === '';
	this.errors.maxEnrollment = this.userInput.maxEnrollment.trim() === '';
	this.errors.enrollment = this.userInput.enrollment.trim() === '';
}
/**
	 * Checks whether there are any validation errors in the user input.
	 * @returns A boolean indicating if there are errors in the input.
	 */
hasErrors(): boolean {
	return Object.values(this.errors).some(x => x);
}
/**
	 * Emits the saveClicked event with the updated user input if there are no validation errors.
	 * This method is called when the user clicks the save button to save changes to the course.
	 */
save(): void {
	this.validate();
	if (this.hasErrors()) return;

	this.saveClicked.emit({
		id: this.selectedCourse?.edit?.id || this.selectedCourse?.id,
		...this.userInput
	});
}
/**
	 * Lifecycle hook that is triggered after the view has been initialized.
	 * This method forces a reflow of the HTML wrapper to ensure correct layout rendering.
	 */
ngAfterViewInit(): void {
	this.forceReflow();
}
/**
	 * Switches the active tab by updating the `tab` property.
	 * This method is called when the user clicks to switch between tabs.
	 * @param next The index of the tab to switch to.
	 */
switchTab(next: number): void {
	this.tab = next;
	this.forceReflow();
}
/**
	 * Forces a reflow of the HTML wrapper by temporarily hiding and showing the element.
	 * This method ensures that the layout is correctly recalculated after certain UI updates.
	 */
private forceReflow(): void {
	setTimeout(() => {
		if (this.htmlWrapperRef?.nativeElement) {
			this.htmlWrapperRef.nativeElement.style.display = 'none';
			this.htmlWrapperRef.nativeElement.offsetHeight;
			this.htmlWrapperRef.nativeElement.style.display = 'block';
		}
	}, 0);
}
}
