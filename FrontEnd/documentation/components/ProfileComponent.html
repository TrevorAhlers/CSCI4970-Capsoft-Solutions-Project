import { Component, Input } from '@angular/core';
import { Router } from '@angular/router';
import { environment } from 'src/environments/environment';
/**
 * ProfileComponent displays the user profile with options to navigate to different routes, 
 * download CSV data, and save changes to the application.
 */
@Component({
	selector: 'app-profile',
	templateUrl: './profile.component.html',
	styleUrls: ['./profile.component.scss']
})
export class ProfileComponent {
	@Input() username: string = 'Guest User';
	profileImage: string = "Capture.png";
	/**
	 * Creates an instance of the ProfileComponent.
	 * @param router The Angular Router to navigate to different views.
	 */
	constructor(private router: Router) {}
/**
 * Navigates to the data page.
 * This method is called when the user clicks a link to view data.
 */
navigateToData() {
	this.router.navigate(['/data']);
}
/**
 * Navigates to the home page.
 * This method is called when the user clicks a link to go to the home page.
 */
navigateToHome() {
	this.router.navigate(['/home']);
}
/**
	 * Navigates to the login page.
	 * This method is called when the user clicks a link to log out or log in.
	 */
navigateToLogin() {
	this.router.navigateByUrl('/login');
}
/**
	 * Downloads the CSV file containing room assignments.
	 * This method is triggered when the user clicks the "Download CSV" button.
	 */
downloadCsv(): void {
	const link = document.createElement('a');
	link.href = `${environment.apiBaseUrl}/api/download`;
	link.download = 'room_assignments.csv';
	link.click();
}
/**
	 * Saves the change log by downloading it as a text file.
	 * This method is triggered when the user clicks the "Save Changes" button.
	 */
saveChanges(): void {
	const link = document.createElement('a')
	link.href      = `${environment.apiBaseUrl}/api/change-log`
	link.download  = 'change_log.txt'
	link.click()
}
}
