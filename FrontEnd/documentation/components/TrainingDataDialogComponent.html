import { Component, Inject } from '@angular/core';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { HttpClient } from '@angular/common/http';
import { environment } from 'src/environments/environment';

interface TrainingInfo {
	files: string[];
}
/**
 * Dialog component for uploading and managing training data files (CSV format).
 */
@Component({
	selector: 'app-training-data-dialog',
	templateUrl: './training-data-dialog.component.html'
})
export class TrainingDataDialogComponent {
	files: string[] = [];
	uploading = false;
	uploadError = '';
	/**
	 * Constructor with dependencies injected.
	 * @param http HttpClient for backend API requests.
	 * @param dialogRef Reference to the current dialog instance.
	 * @param data Data injected into the dialog (unused here).
	 */
	constructor(
		private http: HttpClient,
		public dialogRef: MatDialogRef<TrainingDataDialogComponent>,
		@Inject(MAT_DIALOG_DATA) public data: any
	) {
		this.refresh();
	}
	/**
	 * Refreshes the list of uploaded training files from the backend.
	 */
	private refresh(): void {
		this.http
			.get<TrainingInfo>(`${environment.apiBaseUrl}/training/files`)
			.subscribe(d => (this.files = d.files));
	}
	/**
	 * Handles file upload logic with validation and error feedback.
	 * Only `.csv` files are accepted.
	 * @param event File input change event.
	 */
	uploadFile(event: Event): void {
		this.uploadError = '';
		const file = (event.target as HTMLInputElement).files?.[0];
		if (!file) return;

		if (!file.name.toLowerCase().endsWith('.csv')) {
			this.uploadError = 'Only CSV files are allowed.';
			return;
		}

		const form = new FormData();
		form.append('file', file, file.name);
		this.uploading = true;

		this.http.post(`${environment.apiBaseUrl}/training/upload`, form)
			.subscribe({
				next: () => {
					this.uploading = false;
					this.refresh();
				},
				error: () => {
					this.uploading = false;
					this.uploadError = 'Upload failed. Please try again.';
				}
			});
	}
	/**
	 * Sends a request to reset the training files to their default state.
	 */
	resetDefaults(): void {
		this.http.post(`${environment.apiBaseUrl}/training/reset`, {})
			.subscribe(() => this.refresh());
	}
	/**
	 * Closes the dialog without saving or uploading anything.
	 */
	close(): void {
		this.dialogRef.close();
	}
}
