import { Component, Input, Output, EventEmitter } from '@angular/core';
/**
 * Component that renders a single row of a section table.
 * Emits an event when the row is clicked.
 */
@Component({
	selector: 'app-section-row',
	templateUrl: './section-row.component.html',
	styleUrls: ['./section-row.component.scss']
})
export class SectionRowComponent {
	/**
	 * The data for the row, where keys are column names and values are cell content.
	 */
	@Input() row!: Record<string, any>;
	/**
	 * List of column names to display in the row.
	 */
	@Input() columns: string[] = [];
	/**
	 * Whether this row is currently selected.
	 */
	@Input() selected = false;
	/**
	 * Event emitted when the row is clicked. Sends the row's `id` as payload.
	 */
	@Output() clicked = new EventEmitter<string>();
	/**
	 * Emits the row's `id` through the `clicked` event.
	 */
	emitClick(): void {
		this.clicked.emit(this.row['id']);
	}
}

<div
	class="row row-hover"
	[class.selected]="selected"
	(click)="emitClick()"
	style="cursor: pointer;"
>
	<div
		*ngFor="let col of columns"
		class="cell"
		[ngClass]="{ 'id': col === 'id' }"
	>
		{{ row[col] }}
	</div>
</div>
