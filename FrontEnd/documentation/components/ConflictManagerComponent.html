import { Component, Input, OnChanges, OnInit, OnDestroy, SimpleChanges } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { DataService } from '@services/data.service';
import { Subscription } from 'rxjs';
import { environment } from 'src/environments/environment';
/**
 * Interface representing a conflict view, including the conflict's ID, content, and ignored state.
 */
interface ConflictView {
	id: string;
	content: string;
	ignored: boolean;
}

@Component({
	selector: 'app-conflict-manager',
	templateUrl: './conflict-manager.component.html',
	styleUrls: ['./conflict-manager.component.scss']
})
export class ConflictManagerComponent implements OnInit, OnChanges, OnDestroy {
	@Input() clear = false;

	tab = 0;
	activeConflicts: ConflictView[] = [];
	ignoredConflicts: ConflictView[] = [];
	private activeLoaded = false;
	private ignoredLoaded = false;

	private refreshSub!: Subscription;
	private readonly API = environment.apiBaseUrl;

	constructor(private http: HttpClient, private dataService: DataService) {}
/**
	 * On component initialization, subscribes to conflict refresh events 
	 * and loads both active and ignored conflicts.
	 */
ngOnInit(): void {
	this.refreshSub = this.dataService.conflictRefresh$.subscribe(() => {
		this.reloadBoth();
	});

	this.loadActiveConflicts();
	this.loadIgnoredConflicts();
}
/**
	 * Unsubscribes from the refresh subscription when the component is destroyed.
	 */
ngOnDestroy(): void {
	if (this.refreshSub) this.refreshSub.unsubscribe();
}
/**
	 * Handles changes to the `clear` input property.
	 * If `clear` is true, it resets the conflicts and stops loading them.
	 * Otherwise, it reloads both active and ignored conflicts.
	 * @param changes The changes detected in the input properties.
	 */
ngOnChanges(changes: SimpleChanges): void {
	if (!changes['clear']) return;

	const { previousValue: prev, currentValue: curr } = changes['clear'];
	if (curr === true) {
		this.activeConflicts = [];
		this.ignoredConflicts = [];
		this.activeLoaded = false;
		this.ignoredLoaded = false;
		return;
	}
	if (prev === true && curr === false) this.reloadBoth();
}
/**
	 * Loads the active conflicts from the API if not already loaded or cleared.
	 */
private loadActiveConflicts(): void {
	if (this.clear || this.activeLoaded) return;
	this.http.get<ConflictView[]>(`${this.API}/conflicts/active`)
		.subscribe(r => {
			this.activeConflicts = r ?? [];
			this.activeLoaded = true;
		});
}
/**
	 * Loads the ignored conflicts from the API if not already loaded or cleared.
	 */
private loadIgnoredConflicts(): void {
	if (this.clear || this.ignoredLoaded) return;
	this.http.get<ConflictView[]>(`${this.API}/conflicts/ignored`)
		.subscribe(r => {
			this.ignoredConflicts = r ?? [];
			this.ignoredLoaded = true;
		});
}
/**
	 * Reloads both active and ignored conflicts by resetting load states and fetching data again.
	 */
private reloadBoth(): void {
	this.activeLoaded = false;
	this.ignoredLoaded = false;
	this.loadActiveConflicts();
	this.loadIgnoredConflicts();
}
/**
	 * Handles the action of ignoring a conflict by sending a PUT request to the API.
	 * After ignoring, it reloads both active and ignored conflicts.
	 * @param id The ID of the conflict to be ignored.
	 */
onIgnoreConflict(id: string): void {
	this.http.put(`${this.API}/conflict/ignore/${id}`, {})
		.subscribe(() => this.reloadBoth());
}
/**
	 * Handles the action of restoring a conflict by sending a PUT request to the API.
	 * After restoring, it reloads both active and ignored conflicts.
	 * @param id The ID of the conflict to be restored.
	 */
onRestoreConflict(id: string): void {
	this.http.put(`${this.API}/conflict/activate/${id}`, {})
		.subscribe(() => this.reloadBoth());
}
/**
	 * Switches the active tab between active and ignored conflicts.
	 * @param next The index of the tab to switch to (0 for active conflicts, 1 for ignored conflicts).
	 */
switchTab(next: number): void {
	this.tab = next;
	if (next === 0) this.loadActiveConflicts();
	else this.loadIgnoredConflicts();
}
}
