import {
	Component,
	OnInit,
	AfterViewInit,
	OnDestroy,
	ViewChild,
	Output,
	EventEmitter,
	ChangeDetectorRef
} from '@angular/core';
import { MatTableDataSource } from '@angular/material/table';
import { MatSort } from '@angular/material/sort';
import { HttpClient } from '@angular/common/http';
import { MatDialog } from '@angular/material/dialog';
import { ColumnSelectorDialogComponent } from './column-selector-dialog.component';
import { DataService } from '@services/data.service';
import { Subscription } from 'rxjs';
import { environment } from 'src/environments/environment';
/**
 * Displays a table of course sections with selectable and filterable columns.
 * Allows dynamic column configuration via a modal dialog.
 */
@Component({
	selector: 'app-section-view',
	templateUrl: './section.component.html',
	styleUrls: ['./section.component.scss']
})
export class SectionComponent implements OnInit, AfterViewInit, OnDestroy {
	@ViewChild(MatSort) sort!: MatSort;
	@Output() rowClicked = new EventEmitter<string>();

	courses: any[] = [];
	filteredCourses = new MatTableDataSource<any>();
	displayedColumns: string[] = [];
	allPossibleColumns: string[] = [];

	searchTerm = '';
	selectedId: string | null = null;

	private refreshSub!: Subscription;
	private readonly API = environment.apiBaseUrl;
	/**
	 * Constructor injecting required services.
	 * @param http HttpClient for API communication.
	 * @param cdr ChangeDetectorRef for updating table sorting.
	 * @param dialog MatDialog for opening the column selector.
	 * @param dataService Service to get data and refresh triggers.
	 */
	constructor(
		private http: HttpClient,
		private cdr: ChangeDetectorRef,
		private dialog: MatDialog,
		private dataService: DataService
	) {}
	/**
	 * Angular lifecycle hook. Loads columns and course data.
	 * Subscribes to refresh events.
	 */
	ngOnInit(): void {
		this.fetchAllColumns();
		this.loadCourses();
		this.refreshSub = this.dataService.refresh$.subscribe(() => {
			this.loadCourses();
		});
	}
	/**
	 * Sets the table sorting after view initialization.
	 */
	ngAfterViewInit(): void {
		this.filteredCourses.sort = this.sort;
		this.cdr.detectChanges();
	}
	/**
	 * Cleans up subscriptions on component destroy.
	 */
	ngOnDestroy(): void {
		if (this.refreshSub) {
			this.refreshSub.unsubscribe();
		}
	}
	/**
	 * Fetches the full list of column names available from the backend.
	 */
	private fetchAllColumns(): void {
		this.http.get<any>(`${this.API}/api/columns`).subscribe(data => {
			this.allPossibleColumns = data.columns;
		});
	}
	/**
	 * Loads all course data and sets the default columns to display.
	 */
	loadCourses(): void {
		this.http.get<any>(`${this.API}/api/data`).subscribe(data => {
			this.courses = data.courses;
			this.filteredCourses.data = this.courses;
			if (this.courses.length > 0) {
				const keys = Object.keys(this.courses[0]);
				this.displayedColumns = ['id', ...keys.filter(k => k !== 'id')];
			}
		});
	}
	/**
	 * Loads course data using a selected subset of columns.
	 * @param columns Array of column names to load.
	 */
	loadCoursesWithColumns(columns: string[]): void {
		this.http.post<any>(`${this.API}/api/data/columns`, { columns }).subscribe(data => {
			this.courses = data.courses;
			this.filteredCourses.data = this.courses;
			this.displayedColumns = ['id', ...columns];
		});
	}
	/**
	 * Opens the column selector dialog to customize visible columns.
	 */
	openColumnSelector(): void {
		const dialogRef = this.dialog.open(ColumnSelectorDialogComponent, {
			width: '40rem',
			data: {
				allColumns: this.allPossibleColumns,
				selected: this.displayedColumns.filter(col => col !== 'id')
			}
		});

		dialogRef.afterClosed().subscribe(result => {
			if (result) {
				this.loadCoursesWithColumns(result);
			}
		});
	}
	/**
	 * Filters the table by a search string.
	 * @param value The search term to filter by.
	 */
	applyFilter(value: string): void {
		this.filteredCourses.filter = value.trim().toLowerCase();
	}
	/**
	 * Handles the selection of a table row.
	 * @param courseId The ID of the selected course.
	 */
	handleRowClick(courseId: string): void {
		this.selectedId = courseId;
		this.rowClicked.emit(courseId);
	}
}

	<!-- Container -->

		<mat-divider></mat-divider>

		<!-- Top Tool Bar -->
		<div class="toolbar">
			<button mat-stroked-button class="column-btn" (click)="openColumnSelector()">
				Choose Columns
			</button>

			<mat-form-field appearance="outline" class="search-field">
				<mat-label>Search courses</mat-label>
				<input
					matInput
					placeholder="Type to filter…"
					[(ngModel)]="searchTerm"
					(ngModelChange)="applyFilter($event)"
				/>
				<button
					matSuffix
					mat-icon-button
					*ngIf="searchTerm"
					(click)="searchTerm=''; applyFilter('')"
					aria-label="Clear"
				>
					<mat-icon>close</mat-icon>
				</button>
			</mat-form-field>
		</div>

		<!-- Headers -->
		<div class="header-row">
			<div
				*ngFor="let col of displayedColumns"
				class="header-cell"
				[ngClass]="{ 'id': col === 'id' }"
			>
				{{ col }}
			</div>
		</div>

		<!-- Rows -->
		<div class="row-list-container">
			<app-section-row
				*ngFor="let row of filteredCourses.filteredData"
				[row]="row"
				[columns]="displayedColumns"
				[selected]="row.id === selectedId"
				(clicked)="handleRowClick($event)"
			></app-section-row>
		</div>
	</mat-card>
</div>
