import { Component } from '@angular/core';
import { FormControl, FormGroup, Validators } from '@angular/forms';
import { Router } from '@angular/router';
import { HttpClient } from '@angular/common/http';
import { environment } from 'src/environments/environment';

/**
 * LoginComponent handles user authentication logic.
 * It provides functionality for logging in a user and navigating to the home page upon successful login.
 */
@Component({
	selector: 'app-login',
	templateUrl: './login.component.html',
	styleUrls: ['./login.component.scss']
})
export class LoginComponent {
	/** Flag to indicate whether login attempt failed */
	loginFailed = false;
	/**
	 * Creates an instance of the LoginComponent.
	 * @param router The Angular Router for navigating between pages.
	 * @param http The HttpClient for making API requests.
	 */
	constructor(private router: Router, private http: HttpClient) {}
	/** Form group containing the login form controls */
	loginForm = new FormGroup({
		username: new FormControl('', Validators.required),
		password: new FormControl('', Validators.required)
	});
	/**
	 * Attempts to log the user in using the provided username and password.
	 * On success, stores the JWT token in local storage and navigates to the home page.
	 * On failure, sets the loginFailed flag to true.
	 */
	loginUser() {
		const { username, password } = this.loginForm.value as any;

		this.http.post<{ token: string }>(`${environment.apiBaseUrl}/api/login`, { username, password })
			.subscribe({
				next: res => {
					this.loginFailed = false;
					localStorage.setItem('jwt', res.token);
					this.router.navigateByUrl('/home');
				},
				error: () => {
					this.loginFailed = true;
				}
			});
	}
	/**
	 * Navigates the user to the registration page.
	 */
	navigateToRegistration() {
		this.router.navigateByUrl('/registration');
	}
}
