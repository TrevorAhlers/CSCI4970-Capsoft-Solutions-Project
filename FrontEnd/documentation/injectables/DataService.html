import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable, Subject } from 'rxjs';
import { environment } from 'src/environments/environment';
/**
 * DataService provides methods to interact with the backend API.
 * It manages the courses' data retrieval, updating, and triggering refresh events.
 */
@Injectable({
	providedIn: 'root',
})
export class DataService {
	private refreshTrigger = new Subject<void>();
	refresh$ = this.refreshTrigger.asObservable();
	private _selectedCourse: any = null;

	private conflictRefreshTrigger = new Subject<void>();
	conflictRefresh$ = this.conflictRefreshTrigger.asObservable();
	/**
	 * Creates an instance of the DataService.
	 * @param http The HttpClient for making API requests.
	 */
	constructor(private http: HttpClient) {}
	/**
	 * Retrieves a list of all available courses from the API.
	 * @returns An Observable containing the list of courses.
	 */
	getCourses(): Observable<any> {
		return this.http.get<any>(`${environment.apiBaseUrl}/api/data`);
	}
	/**
	 * Retrieves the details of a specific course by its ID.
	 * @param courseId The ID of the course to fetch details for.
	 * @returns An Observable containing the course details.
	 */
	getCourseDetails(courseId: string): Observable<any> {
		return this.http.get<any>(`${environment.apiBaseUrl}/details/${courseId}`);
	}
	/**
	 * Retrieves editable course data by course ID.
	 * @param courseId The ID of the course to fetch editable data for.
	 * @returns An Observable containing the editable course data.
	 */
	getEditableCourseData(courseId: string): Observable<any> {
		return this.http.get<any>(`${environment.apiBaseUrl}/edit/${courseId}`);
	}
	/**
	 * Saves the edited course data to the backend.
	 * @param courseId The ID of the course being edited.
	 * @param data The data to be saved.
	 * @returns An Observable containing the server's response.
	 */
	saveEditedCourseData(courseId: string, data: any): Observable<any> {
		return this.http.post<any>(`${environment.apiBaseUrl}/edit/save/${courseId}`, data);
	}
	/**
	 * Triggers a refresh event for course-related data.
	 * Other components can subscribe to `refresh$` to react to this event.
	 */
	triggerRefresh(): void {
		this.refreshTrigger.next();
	}
	/**
	 * Triggers a refresh event specifically for conflicting data.
	 * Other components can subscribe to `conflictRefresh$` to react to this event.
	 */
	triggerConflictRefresh(): void {
		this.conflictRefreshTrigger.next();
	}
	/**
	 * Sets the selected course data.
	 * @param course The course data to be set.
	 */
	setCourse(course: any): void {
		this._selectedCourse = course;
	}
	/**
	 * Retrieves the currently selected course data.
	 * @returns The selected course data.
	 */
	getCourse(): any {
		return this._selectedCourse;
	}
	/**
	 * Retrieves the change log from the API.
	 * @returns An Observable containing the change log as a text string.
	 */
	getChangeLog(): Observable<string> {
		return this.http.get(`${environment.apiBaseUrl}/api/change-log`, { responseType: 'text' });
	}
	/**
	 * Retrieves the current logged-in user's username.
	 * @returns An Observable containing the username of the current user.
	 */
	getCurrentUsername(): Observable<{ username: string }> {
		return this.http.get<{ username: string }>(`${environment.apiBaseUrl}/api/current-user`);
	}
}
